39  b2Vec2 callbacks_t::convert_screen_to_world(int32 x, int32 y)
40  {
41  float32 u = x / static_cast<float32>(tw);
42  float32 v = (th - y) / float32(th);
43 
44  float32 ratio = static_cast<float32>(tw) / static_cast<float32>(th);
45  b2Vec2 extents(ratio * 25.0f, 25.0f);
46  extents *= view_zoom;
47 
48  b2Vec2 lower = settings.view_center - extents;
49  b2Vec2 upper = settings.view_center + extents;
50 
51  b2Vec2 p;
52  p.x = (1.0f - u) * lower.x + u * upper.x;
53  p.y = (1.0f - v) * lower.y + v * upper.y;
54  return p;
55  }
