116 {

128 
129  char title[50];

131  main_window = glutCreateWindow(title);
132 
133  //! Here we setup all the callbacks we need
134  //! Some are set via GLUI
135  GLUI_Master.set_glutReshapeFunc(callbacks_t::resize_cb);
136  GLUI_Master.set_glutKeyboardFunc(callbacks_t::keyboard_cb);
137  GLUI_Master.set_glutSpecialFunc(callbacks_t::keyboard_special_cb);
138  GLUI_Master.set_glutMouseFunc(callbacks_t::mouse_cb);
139  //! Others are set directly
140  glutDisplayFunc(callbacks_t::display_cb);
141  glutMotionFunc(callbacks_t::mouse_motion_cb);
142  glutKeyboardUpFunc(callbacks_t::keyboard_up_cb);
143  glutTimerFunc(frame_period, callbacks_t::timer_cb, 0);
144 
145  //! We create the GLUI user interface
146  create_glui_ui();
147 
148  //! Enter the infinite GLUT event loop
149  glutMainLoop();
150 
151  return 0;
